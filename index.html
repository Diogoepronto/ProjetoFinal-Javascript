Crie um programa que gera um número aleatório entre 1 e 6 e pergunte ao utilizador qual é o seu palpite. Mostre a quantidade de palpites certos/errados
(Dica: utilize a Class Math para gerar números aleatórios) 


              <script>
                function novoJogo() {      
                  var numeroAleatorio = Math.floor(Math.random() * 6) + 1;
                  var totalPalpites = 0;
                  
                  while (true) {
                    var palpite = parseInt(prompt("Digite o seu palpite (entre 1 e 6):"));
                  
                    if (isNaN(palpite) || palpite < 1 || palpite > 6) {
                      alert("Palpite inválido! Por favor, digite um número entre 1 e 6.");
                      continue;

                    totalPalpites++;

                    var resultado = verificarPalpite(palpite, numeroAleatorio);

                                                                       "Total de palpites: " + totalPalpites;

              function bubbleSort() {
                var array = [];
                for (var i = 0; i < 100; i++) {
                  array.push(Math.floor(Math.random() * 100));
                }

                var arrayOrdenado = [...array];

                for (var i = 0; i < arrayOrdenado.length - 1; i++) {
                  for (var j = 0; j < arrayOrdenado.length - 1 - i; j++) {
                    if (arrayOrdenado[j] > arrayOrdenado[j + 1]) {
                      var temp = arrayOrdenado[j];
                      arrayOrdenado[j] = arrayOrdenado[j + 1];
                      arrayOrdenado[j + 1] = temp;
                    }
                  }
                }

                var resultado = "<p>Array original:<br> " + array.join(", ") + "</p>" +
                                "<p>Array ordenado:<br> " + arrayOrdenado.join(", ") + "</p>";

              }

Crie um programa que ordene um array de 100 números aleatórios através do algoritmo “Insertion Sort”.
(Dica: https://www.youtube.com/watch?v=DFG-XuyPYUQ&t=406s) 


              <p id="resultado"></p>

              <button onclick="insertionSort()">Ordenar array</button>

              <script>
              function insertionSort() {
                var array = [];
                for (var i = 0; i < 100; i++) {
                  array.push(Math.floor(Math.random() * 100));
                }

                var arrayOrdenado = [...array];

                for (var i = 1; i < arrayOrdenado.length; i++) {
                  var key = arrayOrdenado[i];
                  var j = i - 1;

                  while (j >= 0 && arrayOrdenado[j] > key) {
                    arrayOrdenado[j + 1] = arrayOrdenado[j];
                    j = j - 1;
                  }

                  arrayOrdenado[j + 1] = key;
                }

                var resultado = "<p>Array original:<br> " + array.join(", ") + "</p>" +
                                "<p>Array ordenado:<br> " + arrayOrdenado.join(", ") + "</p>";

                document.getElementById("resultado").innerHTML = resultado;
              }
              </script>
